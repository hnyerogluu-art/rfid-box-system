  <p>Item Count: <span id="count">0</span></p>
  <p class="status">Status: <span id="status">EMPTY</span></p>

  <button onclick="addItem()">RFID Tag Added</button>
  <button onclick="reset()">Reset</button>
</div>

<script>
let count = 0;

function updateStatus() {
  document.getElementById("count").innerText = count;

  let status = "";

  if (count === 0) {
    status = "EMPTY";
  } else if (count <= 3) {
    status = "PARTIAL";
  } else {
    status = "FULL";
  }

  document.getElementById("status").innerText = status;
}

function addItem() {
  count++;
  updateStatus();
}

function reset() {
  count = 0;
  updateStatus();
}

updateStatus();
</script>
