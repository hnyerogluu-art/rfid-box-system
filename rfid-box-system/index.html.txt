  <p>Item Count: <span id="count">0</span></p>
  <p class="status">Status: <span id="status">EMPTY</span></p>

  <button onclick="addItem()">RFID Tag Added</button>
  <button onclick="reset()">Reset</button>
</div>

<script>
let count = 0;

function update() {
  document.getElementById("count").innerText = count;

  if (count === 0) {
    document.getElementById("status").innerText = "EMPTY";
  } 
  else if (count <= 3) {
    document.getElementById("status").innerText = "PARTIAL";
  } 
  else {
    document.getElementById("status").innerText = "FULL";
  }
}

function addItem() {
  count++;
  update();
}

function reset() {
  count = 0;
  update();
}

update();
</script>
